131 static unsigned long PLLMNCal(unsigned long Fin,
132  unsigned long Fout, unsigned long * pM, unsigned long *pN, unsigned long *pDiv)
133 {
134  unsigned long M = 0; // PLL Multiplier Value
135  unsigned long N = 0; // PLL Divider Value
136  unsigned long Fcco = 0; // PLL Fcco Output frequency
137  unsigned long Fclk = 0; // System Clock frequency
138  unsigned long i = 0; // Counter
139 
140  /************ Check Input parameters ****************/
141  // Input Frequency Range: 1 --> 25 Mhz
142  if((Fin < 1000000) || (Fin > 25000000))
143  {
144  return (0);
145  }
146 
147  // Target Maximum Frequency Range: 10 --> 120Mhz
148  if( (Fout == 0) || (Fout >= 120000000) )
149  {
150  return (0);
151  }
152 
153  // Check pN/pM Pointer valid
154  if((pN == 0) || (pM == 0))
155  {
156  return (0);
157  }
158 
159  /************ Calculate Multiplier/Divider **********/
160  for(N = 1; N < 32; N++)
161  {
162 
163  // The value of PLOCK0 may not be stable when the PLL reference frequency
164  // (FREF, the frequency of REFCLK, which is equal to the PLL input
165  // frequency divided by the pre-divider value) is less than 100 kHz or
166  // greater than 20 MHz
167  Fcco = Fin/N;
168  if((Fcco < 100000) || (Fcco > 20000000))
169  {
170  continue;
171  }
172 
173  for(M = 6; M < 512; M++)
174  {
175  // Fcco Range: 275 --> 550 MHz
176  Fcco = Fin/N;
177  Fcco = 2*Fcco*M;
178  if( (Fcco < 275000000) || (Fcco > 550000000) )
179  {
180  continue;
181  }
182 
183  for(i = 1; i < 256; i++)
184  {
185  Fclk = Fcco/i;
186  if(Fclk > 120000000)
187  {
188  continue;
189  }
190 
191  // Fclk and Fout Maximum different is 10K
192  if( ((Fclk - Fout) <= 10000) || ((Fout - Fclk) <= 10000) )
193  {
194  // Find suitable Multiplier/Divider value.
195  *pM = M;
196  *pN = N;
197  *pDiv = i;
198  return (1);
199  }
200  }
201 
202  }
203  }
204 
205  // Can not find suitable Multiplier/Divider value.
206  return (0);
207 }

544 void SysCtlExtIntCfg(unsigned long ulPin, unsigned long ulCfg)
545 {
546  unsigned long i = 0;
547  unsigned long ulTmpReg1 = 0;
548  unsigned long ulTmpReg2 = 0;
549 
550  // Check input parameters
551  xASSERT( (ulPin & EXT_INT_MASK) != 0 );
552  xASSERT( (ulCfg & EXT_INT_MASK) != 0 );
553 
554  ulTmpReg1 = xHWREG(EXTMODE); // External Interrupt Mode register
555  ulTmpReg2 = xHWREG(EXTPOLAR); // External Interrupt Polar register
556 
557  for(i = 0; i < 4; i++)
558  {
559  if(ulPin & (0x01 << i))
560  {
561  switch(ulCfg)
562  {
563  case EXT_INT_LV_H: // High Level Triggle
564  {
565  ulTmpReg1 &= ~(0x01 << i);
566  ulTmpReg2 |= (0x01 << i);
567  break;
568  }
569  case EXT_INT_LV_L: // Low Level Triggle
570  {
571  ulTmpReg1 &= ~(0x01 << i);
572  ulTmpReg2 &= ~(0x01 << i);
573  break;
574  }
575  case EXT_INT_EG_R: // Rising Triggle
576  {
577  ulTmpReg1 |= (0x01 << i);
578  ulTmpReg2 |= (0x01 << i);
579  break;
580  }
581  case EXT_INT_EG_F: // Falling Triggle
582  {
583  ulTmpReg1 |= (0x01 << i);
584  ulTmpReg2 &= ~(0x01 << i);
585  break;
586  }
587  }
588  }
589  }
590 
591  // Write back to mode/polar register
592  xHWREG(EXTMODE) = ulTmpReg1; // Mode register
593  xHWREG(EXTPOLAR) = ulTmpReg2; // Polar register
594 
595 }
