244  // In the Uart IP block, baud rate is calculated using FDR and DLL-DLM registers
245  // The formula is :
246  // BaudRate= uPeriClk * (mulFracDiv/(mulFracDiv+dividerAddFracDiv) / (16 * (DLL)
247  // It involves floating point calculations. That's the reason the formulae are adjusted with
248  // Multiply and divide method.
249  // The value of mulFracDiv and dividerAddFracDiv should comply to the following expressions:
250  // 0 < mulFracDiv <= 15, 0 <= dividerAddFracDiv <= 15
251 
252  // Worst case
253  best_error = 0xFFFFFFFF;
254  bestd = 0;
255  bestm = 0;
256  best_divisor = 0;
257 
258  for (m = 1 ; m <= 15 ;m++)
259  {
260  for (d = 0 ; d < m ; d++)
261  {
262  divisor = ((unsigned long long)ulPeriClk<<28)*m/(ulBaudrate*(m+d));
263  current_error = divisor & 0xFFFFFFFF;
264 
265  tmp = divisor>>32;
266 
267  // Adjust error
268  if(current_error > ((unsigned long)1<<31))
269  {
270  current_error = -current_error;
271  tmp++;
272  }
273 
274  if(tmp<1 || tmp>65536)
275  {
276  continue;
277  }
278 
279  if( current_error < best_error)
280  {
281  best_error = current_error;
282  best_divisor = tmp;
283  bestd = d;
284  bestm = m;
285  if(best_error == 0)
286  {
287  break;
288  }
289  }
290  }
291 
292  if (best_error == 0)
293  {
294  break;
295  }
296  }
297 
298  // Can not find best match
299  if(best_divisor == 0)
300  {
301  return xfalse;
302  }
303 
304  recalcbaud = (ulPeriClk>>4) * bestm/(best_divisor * (bestm + bestd));
305 
306  // Reuse best_error to evaluate baud error
307  if(ulBaudrate > recalcbaud)
308  {
309  best_error = ulBaudrate - recalcbaud;
310  }
311  else
312  {
313  best_error = recalcbaud -ulBaudrate;
314  }
315 
316  best_error = best_error * 100 / ulBaudrate;
317 
318  if (best_error < max_error)
319  {
320  if(ulBase == UART1_BASE)
321  {
322  // Nothing to do Now.
323  }
324  else
325  {
326  //Configure DLM/DLL
327  xHWREG(ulBase + LCR) |= LCR_DLAB;
328  xHWREG(ulBase + DLM) = (best_divisor >> 8) & 0xFF;
329  xHWREG(ulBase + DLL) = (best_divisor >> 0) & 0xFF;
330 
331  // DLAB Must be zero.
332  xHWREG(ulBase + LCR) &= ~LCR_DLAB;
333  bestd &= (unsigned long)0x0F;
334  bestm = (bestm << 4) & (unsigned long)0xF0;
335  xHWREG(ulBase + FDR) = bestd | bestm;
336  }
337 
338  return (xtrue); // Success configure Baud registers.
339  }
340 
341  return (xfalse); // Failure to configure Baud registers.
342 }
