97 void ADCInit(unsigned long ulBase, unsigned long ulRate)
98 {
99 
100  unsigned long temp = 0;
101  unsigned long ulClk = 0;
102 
103  // Note: ADC Module Maximum frequecy is 13MHz
104  // ulRate <= 200KHz
105  // ulClk = SysCtlPeripheralClockGet(SYSCTL_PERIPH_ADC);
106  // ulClk /= 13000000;
107 
108  //xHWREG(ulBase + AD_CR) = (ulClk<<CR_CLKDIV_S) | BIT_32_0;
109  xHWREG(ulBase + AD_CR) = 0;
110 
111  // Set clock frequency
112  ulClk = SysCtlPeripheralClockGet(SYSCTL_PERIPH_ADC);
113 
114  // The APB clock (PCLK_ADC0) is divided by (CLKDIV+1) to produce the clock for
115  // A/D converter, which should be less than or equal to 13MHz.
116  // A fully conversion requires 65 of these clocks.
117  // ADC clock = PCLK_ADC0 / (CLKDIV + 1);
118  // ADC rate = ADC clock / 65;
119  temp = ulRate * 65;
120 
121  // Get the round value by fomular: (2*A + B)/(2*B)
122  temp = (ulClk * 2 + temp)/(2 * temp) - 1;
123 
124  xHWREG(ulBase + AD_CR) = (temp<<CR_CLKDIV_S) | BIT_32_0;
125 }
